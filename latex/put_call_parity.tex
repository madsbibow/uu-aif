\Large
\textbf{Put-Call Parity}

\vspace{1em}

\huge
\[
C + \frac{X}{(1+r_f)^T} = P + S_0
\]

\vspace{0.5em}

\normalsize
\begin{tabular}{rl}
$C$ & = Call premium \\
$P$ & = Put premium \\
$S_0$ & = Spot (stock) price \\
$X$ & = Strike price \\
$r_f$ & = Risk-free rate \\
$T$ & = Time to maturity
\end{tabular}
