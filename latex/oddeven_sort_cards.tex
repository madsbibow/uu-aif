% Card dimensions for 2x4 grid on A4
\newcommand{\cardwidth}{10.2cm}
\newcommand{\cardheight}{7.25cm}

\newcommand{\card}[1]{%
  \begin{tikzpicture}
    \draw[thick] (0,0) rectangle (\cardwidth, \cardheight);
    \node at (0.5*\cardwidth, 0.55*\cardheight) {%
      \fontsize{72}{80}\selectfont\bfseries #1%
    };
    \node at (0.5*\cardwidth, 0.15*\cardheight) {%
      \fontsize{22}{26}\selectfont\bfseries%
      \ifodd#1 ODD\else EVEN\fi%
    };
  \end{tikzpicture}%
}

\newcommand{\cardpage}[8]{%
\begin{tabular}{@{}c@{}c@{}}

\card{#3} & \card{#4} \\
\card{#5} & \card{#6} \\
\card{#7} & \card{#8}
\end{tabular}%
\newpage}
